CREATE SCHEMA IF NOT EXISTS tienda;

CREATE TABLE IF NOT EXISTS tienda.administradores (
	id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
        email TEXT NOT NULL UNIQUE,
        password TEXT NOT NULL
);

CREATE SEQUENCE IF NOT EXISTS ventas.catalogo_id_seq
    INCREMENT 1
    START 1
    MINVALUE 1
    MAXVALUE 2147483647
    CACHE 1;

CREATE TABLE IF NOT EXISTS ventas.catalogo (
    id integer NOT NULL DEFAULT nextval('ventas.catalogo_id_seq'::regclass),
    name text COLLATE pg_catalog."default" DEFAULT 'Producto'::text,
    description text COLLATE pg_catalog."default" DEFAULT 'Una descripcion elocuente y util para el usuario'::text,
    category text COLLATE pg_catalog."default" DEFAULT 'Misc.'::text,
    price integer DEFAULT 0,
    rating real DEFAULT 0,
    image text DEFAULT '/products/placeholder.png'::text,
    CONSTRAINT catalogo_pk PRIMARY KEY (id)
)
